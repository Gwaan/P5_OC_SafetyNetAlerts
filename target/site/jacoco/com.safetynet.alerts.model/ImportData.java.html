package com.safetynet.alerts.model;

public class ImportData {

    /**
     * The Persons.
     */
    List<Person> persons;

    /**
     * The Fire stations.
     */
    @JsonProperty("firestations")

    /**
     * The Medical records.
     */

    private List<MedicalRecord> medicalRecords;


}
